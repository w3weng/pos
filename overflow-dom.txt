        (async function () {
            const output = document.getElementById('output');
            const timestamp = Date.now();
            const baseUrl = `${window.location.protocol}//${window.location.hostname}:5000/api`;
            const email = `overflow-test-${timestamp}@example.com`;
            const password = 'Password123!';

            const postJson = async (path, payload, token) => {
                const response = await fetch(`${baseUrl}${path}`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        ...(token ? { Authorization: `Bearer ${token}` } : {}),
                    },
                    body: JSON.stringify(payload),
                });
                const data = await response.json();

                if (!response.ok || !data.success) {
                    throw new Error(data.message || `Request failed: ${path}`);
                }

                return data.data;
            };

            try {
                await postJson('/auth/register', {
                    storeName: `Overflow Test ${timestamp}`,
                    email,
                    password,
                    phone: '09170000000',
                });

                const loginData = await postJson('/auth/login', { email, password });
                const { token, user } = loginData;

                localStorage.setItem('token', token);
                localStorage.setItem('user', JSON.stringify(user));
                localStorage.setItem('auth-store', JSON.stringify({
                    state: {
                        user,
                        token,
                        isAuthenticated: true,
                    },
                    version: 0,
                }));

                const category = await postJson('/categories', {
                    name: 'Overflow Test Category',
                    description: 'Temporary category for layout debugging',
                }, token);

                await postJson('/products', {
                    name: 'Overflow Test Product',
                    description: 'Temporary product for layout debugging',
                    categoryId: String(category.id),
                    sku: `OVERFLOW-${timestamp}`,
                    barcode: `${timestamp}`,
                    purchasePrice: '50',
                    sellingPrice: '75',
                    imageUrl: '',
                    isAvailable: true,
                }, token);

                const iframe = document.createElement('iframe');
                iframe.src = '/pos';
                iframe.style.width = '390px';
                iframe.style.height = '844px';
                iframe.style.border = '0';
                document.body.appendChild(iframe);

                iframe.onload = () => {
                    setTimeout(() => {
                        try {
                            const doc = iframe.contentDocument;
                            const win = iframe.contentWindow;
                            const viewportWidth = win.innerWidth;
                            const scrollWidth = doc.documentElement.scrollWidth;

                            const overflowElements = Array.from(doc.querySelectorAll('body *'))
                                .map((element) => {
                                    const rect = element.getBoundingClientRect();
                                    return {
                                        tag: element.tagName.toLowerCase(),
                                        className: element.className || '',
                                        text: (element.textContent || '').trim().slice(0, 80),
                                        left: Math.round(rect.left),
                                        right: Math.round(rect.right),
                                        width: Math.round(rect.width),
                                    };
                                })
                                .filter((item) => item.right > viewportWidth + 1 || item.left < -1)
                                .slice(0, 25);

                            output.textContent = JSON.stringify({
                                viewportWidth,
                                scrollWidth,
                                overflowElements,
                            }, null, 2);
                        } catch (error) {
                            output.textContent = error.message || 'Failed to inspect iframe';
                        }
                    }, 4000);
                };
            } catch (error) {
                output.textContent = error.message || 'Mobile overflow debug failed';
            }
        }());
    </script>
